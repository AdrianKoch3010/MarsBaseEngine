222  void AStarPathfinder<TTileMap>::ExpandNode(detail::AStarNode::Ptr currentNode, Position endPosition)
223  {
224  for (const auto & succeedingNode : this->GetNeighbours(currentNode->position))
225  {
226  // If the succeeding node is already on the closed list, continue
227  if (succeedingNode->closed)
228  continue;
229 
230  // The movement cost to that node --> maybe add something more fancy later on
231  auto edgeValue = map.GetTileMovementSpeed(succeedingNode->position.x, succeedingNode->position.y);
232 
233  // Calculate the cost for the new route: g value of the previous node + cost for edge / movement cost to the current node
234  auto tentativeG = currentNode->g + edgeValue;
235 
236  // If the succeding node is already on the open list but the new route isn't better that the existing one, do nothing
237  if (succeedingNode->open && tentativeG >= succeedingNode->g)
238  continue;
239 
240  // If not -> Set the previous node field and store the g value as the new best value
241  succeedingNode->g = tentativeG;
242  succeedingNode->previousNode = currentNode;
243  succeedingNode->h = succeedingNode->DistanceTo(endPosition);
244 
245  // If the succeeding node isn't on the openList yet, add it to the open list
246  if (!succeedingNode->open)
247  {
248  succeedingNode->open = true;
249  openList.push(succeedingNode);
